22       * @return the number of raters

28      /**
29       * @return the average rating for this ID if there are at least minimalRaters ratings
30       * @param id
31       * @param minimalRaters
32       * @return
33       */
34      private double getAverageByID(String id,int minimalRaters){
35          double average =0;
36          double total=0;
37          int countRaters= 0;
38          for(Rater rater:myRaters){
39              if(rater.hasRating(id)){
40                  countRaters++;
41                  total = total + rater.getRating(id);
42              }
43          }
44          
45          if(countRaters >= minimalRaters){
46              average = total/countRaters;
47          }
48          return average;
49          
50      }
51  
52      /**
53       * @param minimalRaters
54       * @return an ArrayList of all the Rating objects for movies that have at least the minimal number of raters supplying a rating
55       */
56      public ArrayList<Rating> getAverageRatings(int minimalRaters){
57          ArrayList<Rating> avgRatingList = new ArrayList<Rating>();
58          ArrayList<String> movies = MovieDatabase.filterBy(new TrueFilter());
59          for(String id:movies){
60              double avg = getAverageByID(id,minimalRaters);
61              if(avg>0.0){
62                  Rating currRating = new Rating(id,avg);
63                  avgRatingList.add(currRating);
64              }    
65          }
66  
67          return avgRatingList;
68      }

75      public ArrayList<Rating>  getAverageRatingsByFilter(int minimalRaters,Filter filterCriteria)
